namespace = nights_watch

# nights_watch.1 - hands out the black_brother trait to rulers (baron and up)
# nights_watch.2 - expels children and spouses from the NW
# nights_watch.3 - randomises culture and religion of characters on gaining titles in the NW
# nights_watch.4 - call the northern lords to the Wall defense
# nights_watch.5 - handing out commands

# Insignificant impact, fired on new holder and twice a year.
character_event = {
    id = nights_watch.1
    
    hide_window = yes
    is_triggered_only = yes
    
    min_age = 16
    
    trigger = {
        OR = {
            is_married = yes
            NOT = {
                trait = black_brother
            }
        }
        OR = {
            primary_title = {
                OR = {
                    title = d_the_wall
                    title = k_the_nights_watch
                }
            }
            any_liege = {
                primary_title = {
                    OR = {
                        title = d_the_wall
                        title = k_the_nights_watch
                    }
                }
            }
        }
    }
    
    immediate = {
		set_character_flag = no_court_invites
        if = {
            limit = {
				is_female = no
				age = 16
				is_tribal = no
                NOT = { trait = black_brother }
				NOT = { liege = { is_tribal = yes } }
            }
            add_trait = black_brother
			add_trait = cannot_marry
			set_character_flag = no_court_invites
        }
        if = {
            limit = {
                is_married = yes
            }
            remove_spouse = spouse
        }
		any_realm_lord = {
			if = {
				limit = { 
					trait = black_brother
					NOR = {
						has_landed_title = c_easttown
						has_landed_title = c_moles_town
						has_landed_title = c_northtown
						has_landed_title = c_queenscrown
						has_landed_title = c_westtown
					}
				}
				set_government_type = nights_watch_government
				primary_title = { succession = appointment }
				recalc_succession = yes
			}
			if = {
				limit = {
					OR = {
						has_landed_title = c_easttown
						has_landed_title = c_moles_town
						has_landed_title = c_northtown
						has_landed_title = c_queenscrown
						has_landed_title = c_westtown
					}
					tsk_is_nights_watch_trigger = yes
				}
				holder_scope = {
					remove_trait = black_brother
					remove_trait = nw_steward
					remove_trait = nw_ranger
					remove_trait = nw_builder
					set_government_type = tribal_government
					primary_title = { succession = gavelkind }
					recalc_succession = yes
					culture = hill_clansmen
					set_graphical_culture = hill_clansmen
					religion = old_gods
					health = 8
					fertility = 0.8
				}
			}
		}
	}
}

# Court cleanup
character_event = {
    id = nights_watch.2
    
    hide_window = yes
	only_playable = yes
    is_triggered_only = yes
    
    trigger = {
        tsk_is_nights_watch_trigger = yes
		is_landed = yes
    }
    
	immediate = {
		any_courtier = {
			limit = {
				tsk_is_not_nights_watch_trigger = yes
			}
			death = { death_reason = death_murder_unknown }
		}
	}
}


# nights_watch.3 - randomises the order of a NW brother
character_event = {
    id = nights_watch.3
    
    hide_window = yes
    is_triggered_only = yes
	only_men = yes
    
    trigger = {
        trait = black_brother
		any_liege = { trait = black_brother }
        NOR = {
            trait = nw_ranger
			trait = nw_builder
			trait = nw_steward
		}
    }
    
    immediate = {
	
        random_list = { #Rangers
            33 = {
				modifier = {
					factor = 0
					martial <= 9
				}
				modifier = {
					factor = 0
					tsk_cant_work_trigger = yes
				}
				modifier = {
					factor = 0.5
					trait = craven
				}
				modifier = {
					factor = 5
					martial = 12
				}
				modifier = {
					factor = 4
					is_strong_trigger = yes
				}
				modifier = {
					factor = 3
					OR = { 	
						trait = misguided_warrior
						trait = tough_soldier
						trait = skilled_tactician
						trait = brilliant_strategist
					}
				}
				modifier = {
					factor = 3
					OR = { 	
						trait = dueling_novice
						trait = dueling_adept
						trait = dueling_skilled
						trait = dueling_expert
						trait = dueling_master
						trait = dueling_legend
					}
				}
				modifier = {
					factor = 2
					father_even_if_dead = { higher_tier_than = COUNT }				
				}
				modifier = {
					factor = 1.5
					trait = hunter
				}
				modifier = {
					factor = 1.5
					trait = brave
				}
				modifier = {
					factor = 1.5
					trait = tall
				}
				modifier = {
					factor = 1.5
					trait = burly
				}
				add_trait = nw_ranger
			}
            33 = { #Builders
				modifier = {
					factor = 0
					stewardship <= 9
				}
				modifier = {
					factor = 0
					tsk_cant_work_trigger = yes
				}
				modifier = {
					factor = 0.5
					trait = slothful
				}
				modifier = {
					factor = 5
					stewardship = 12
				}
				modifier = {
					factor = 1.5
					has_virtue_trigger = yes
				}
				modifier = {
					factor = 3
					OR = { 	
						trait = indulgent_wastrel
						trait = thrifty_clerk
						trait = fortune_builder
						trait = midas_touched
					}
				}
				add_trait = nw_builder
			}
            33 = { #Stewards
				modifier = {
					factor = 0.5
					is_dumb_trigger = yes
				}
				modifier = {
					factor = 5
					is_smart_trigger = yes
				}
				modifier = {
					factor = 2
					father_even_if_dead = { higher_tier_than = COUNT }				
				}
				add_trait = nw_steward
			}
        }
    }
}

character_event = {
    id = nights_watch.999
    
    hide_window = yes
    is_triggered_only = yes

	immediate = {
		if = {
			limit = {
				is_ruler = yes
			}
			current_heir = { save_event_target_as = wealthtransfer_nw }
		}
		else_if = {
			limit = {
				is_ruler = no
			}
			host = { save_event_target_as = wealthtransfer_nw }
		}
		banish = yes
	}
	
	option = {
		if = {
			limit = {
				is_in_society = yes
			}
			leave_society = yes
		}
		if = {
			limit = {
				is_ruler = no
			}
			transfer_scaled_wealth = {
				to = event_target:wealthtransfer_nw
				value = all
			}
		}
		any_spouse = { remove_spouse = PREV }	
		any_consort = { remove_consort = PREV }
		any_lover = { remove_lover = PREV }
		add_trait = black_brother		
		set_character_flag = no_court_invites
		any_claim = {
			remove_claim = PREV
		}
		k_the_nights_watch = {
			holder_scope = {
				ROOT = { move_character = PREV }
			}
		}
	}
}
